def entropy(signal):
        '''
        function returns entropy of a signal
        signal must be a 1-D numpy array
        '''
        lensig=signal.size
        symset=list(set(signal))
        propab=[np.size(signal[signal==i])/(1.0*lensig) for i in symset]
        ent=np.sum([p*np.log2(1.0/p) for p in propab])
        return ent

def entropy_heatmap(image, k=5):
    '''
    image: 2d numpy array as an input
    k (int) : number of neighbours
    '''
    S=image.shape
    E=np.array(image)
    for row in range(S[0]):
        for col in range(S[1]):
            Lx=np.max([0,col-k])
            Ux=np.min([S[1],col+k+1]) # +1 due to x[a:b] -> [a;b)
            Ly=np.max([0,row-k])
            Uy=np.min([S[0],row+k+1])
            region=image[Ly:Uy,Lx:Ux].flatten()
            E[row,col]=entropy(region)
    return E

def digitize(image, bins=10):
    _, bin_edges = np.histogram([0,255], bins=bins)
    digitized = np.digitize(image, bin_edges)
    return digitized

Z analizy heat map wynika, e najwiksza rnica jest w obszarach gdzie kolor jest mniej wicej stay, domylam si e HSV nieco lepiej radzi sobie z szumem, poniewa w RGB kada warstwa posiada sta czstke szumu niezalenie od koloru, a jeeli mamy np. zielon traw to warstwa H w HSV prawdopodobnie ma bardzo nisk entropi - co nie wystpuje w RGB poniewa warstwa G bdzie miaa te elementy szumu pochodzce z rnic w jasnoci pikseli. Podobnie np. woda lub mga na warstwie odpowiedzialnej za jasno bdzie miaa nisk entropi. Z analizy rozkadw poszczeglnych entropii w zalenoci od iloci ssiadw widzimy, e ronie ona gdy ilo ssiadw jest wiksza w podobny sposb zarwno dla RGB jak i HSV. Mediana ze rednich entropii jest zawsze mniejsza dla HSV.
